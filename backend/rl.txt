✅ Step 3: Define RL Environment

Use Gym-style environment:

import gym
from gym import spaces
import numpy as np

class PromptEnv(gym.Env):
    def __init__(self, data):
        super().__init__()
        self.data = data
        self.index = 0
        
        self.action_space = spaces.Discrete(4)  # allow, block, sanitize, ask
        self.observation_space = spaces.Box(low=-1, high=1, shape=(384,), dtype=np.float32)

    def reset(self):
        self.index = np.random.randint(len(self.data))
        prompt = self.data[self.index]["prompt"]
        return get_state(prompt)

    def step(self, action):
        sample = self.data[self.index]
        label = sample["label"]
        
        reward = self.get_reward(action, label)
        
        done = True
        next_state = np.zeros(384)
        
        return next_state, reward, done, {}

    def get_reward(self, action, label):
        # action: 0 allow, 1 block, 2 sanitize, 3 ask
        
        if label == "safe":
            if action == 0:
                return +1
            else:
                return -0.5
        
        if label == "malicious":
            if action in [1,2,3]:
                return +1
            else:
                return -1

✅ Step 5: Inference (Defense Layer)
model = PPO.load("prompt_defense_agent")

def defense_decision(prompt):
    state = get_state(prompt)
    action, _ = model.predict(state)

    actions = ["ALLOW", "BLOCK", "SANITIZE", "ASK"]
    return actions[action]
✅ Step 6: Sanitization Logic
def sanitize_prompt(prompt):
    blocked_words = ["ignore previous", "system prompt", "reveal"]
    
    for word in blocked_words:
        prompt = prompt.replace(word, "[FILTERED]")
    
    return prompt

🔁 Step 8: Full Pipeline
def pipeline(user_prompt):
    decision = defense_decision(user_prompt)
    
    if decision == "BLOCK":
        return "❌ Prompt blocked due to security risk"
    
    elif decision == "SANITIZE":
        clean = sanitize_prompt(user_prompt)
        return call_llm(clean)
    
    elif decision == "ASK":
        return "⚠️ Prompt suspicious. Confirm?"
    
    else:
        return call_llm(user_prompt)

    st.write("### Decision:", decision)
    
    if decision == "ALLOW":
        response = call_llm(prompt)
        st.write("### Response:", response)
    
    elif decision == "BLOCK":
        st.error("Blocked: Potential prompt injection detected")
    
    elif decision == "SANITIZE":
        clean = sanitize_prompt(prompt)
        st.write("Sanitized Prompt:", clean)
        response = call_llm(clean)
        st.write("Response:", response)
    
    elif decision == "ASK":
        st.warning("Suspicious prompt. Please confirm.")
